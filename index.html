async function sendNotification() { const type = document.getElementById('notificationType').value; const title = document.getElementById('notificationTitle').value.trim(); const message = document.getElementById('notificationMessage').value.trim(); const icon = document.getElementById('notificationIcon').value.trim() || '📢'; const target = document.getElementById('notificationTarget').value; if (!title || !message) { showMessage('заполните заголовок и сообщение', 'error'); return; } try { // Создаём уведомление const { data: notification, error } = await supabaseClient .from('notifications') .insert({ type: type, title: title, message: message, icon: icon, created_by: 1, // ID админа target_all: target === 'all', created_at: new Date().toISOString() }) .select() .single(); if (error) throw error; if (target === 'all') { // Получаем всех пользователей const { data: users } = await supabaseClient .from('users') .select('id'); // Создаём связи для всех пользователей const userNotifications = users.map(u => ({ user_id: u.id, notification_id: notification.id })); await supabaseClient .from('user_notifications') .insert(userNotifications); showMessage(`✅ уведомление отправлено ${users.length} пользователям`, 'success'); } else { // Отправляем конкретному пользователю const userId = document.getElementById('notificationUserId').value; if (!userId) { showMessage('введите ID пользователя', 'error'); return; } await supabaseClient .from('user_notifications') .insert({ user_id: parseInt(userId), notification_id: notification.id }); showMessage(`✅ уведомление отправлено пользователю ${userId}`, 'success'); } // Очищаем форму document.getElementById('notificationTitle').value = ''; document.getElementById('notificationMessage').value = ''; document.getElementById('notificationIcon').value = '📢'; } catch (error) { showMessage('ошибка: ' + error.message, 'error'); } }
